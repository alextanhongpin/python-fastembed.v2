from ollama import ChatResponse, chat


def query_deepseek(prompt) -> str:
    response: ChatResponse = chat(
        model="deepseek-r1:8b",
        messages=[
            {
                "role": "user",
                "content": prompt,
            },
        ],
    )
    return response.message.content
